\begin{frame}{Install a Package}
  \begin{itemize}
    \item Most packages are already installed on Linux and Mac
    \item For Ubuntu: \texttt{texlive-full} package provides all available packages
    \item On Windows with MiKTeX, packages are downloaded automatically when included in the document
  \end{itemize}
\end{frame}

\begin{frame}{Purpose of Packages}
  \begin{itemize}
    \item Packages extend the functionality of LaTeX
    \item Different packages serve different purposes
    \item Explaining useful packages in the tutorials
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Using/Including a Package}
  \begin{itemize}
    \item To import a package, add \texttt{\textbackslash{}usepackage\{PACKAGENAME\}} to the preamble
    \item Example code:
    \begin{verbatim}
\documentclass{article}
\usepackage{PACKAGENAME}
\begin{document}
...
\end{document}
    \end{verbatim}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Math Typesetting with \texttt{amsmath} Package}
  \begin{itemize}
    \item \texttt{amsmath} package provides math typesetting functionality
    \item Use the \texttt{equation} environment for math equations
    \item Automatic equation numbering
    \item Example code:
    \begin{verbatim}
\documentclass{article}
\begin{document}
\begin{equation}
  f(x) = x^2
\end{equation}
\end{document}
    \end{verbatim}
    \item Output: $f(x) = x^2$ \hfill (1)
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Removing Equation Numbers}
  \begin{itemize}
    \item Sometimes it's necessary to remove equation numbers
    \item Use the \texttt{equation*} environment
    \item Include the \texttt{amsmath} package for this functionality
    \item Example code:
    \begin{verbatim}
\documentclass{article}
\usepackage{amsmath}
\begin{document}
\begin{equation*}
  f(x) = x^2
\end{equation*}
\end{document}
    \end{verbatim}
    \item Output: $f(x) = x^2$
  \end{itemize}
\end{frame}

\begin{frame}{Conclusion}
  \begin{itemize}
    \item Packages enhance LaTeX functionality
    \item Import packages using \texttt{\textbackslash{}usepackage\{PACKAGENAME\}}
    \item Example package: \texttt{amsmath} for math typesetting
    \item Use \texttt{equation} environment for numbered equations
    \item Use \texttt{equation*} environment for equations without numbers
  \end{itemize}
\end{frame}
