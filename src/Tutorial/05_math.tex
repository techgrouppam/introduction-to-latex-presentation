\begin{frame}{Inline Math}
  \begin{itemize}
    \item Two modes of typesetting math in LaTeX
    \item Embedding math directly into text using dollar signs
    \item Example: \texttt{This formula \$f(x) = x\^{}2\$ is an example.}
    \item Output equation: This formula $f(x) = x^2$ is an example.
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{The Equation and Align Environment}
  \begin{itemize}
    \item Use \texttt{equation} environment for single equations
    \item Use \texttt{align} environment for multiple equations and automatic alignment
    \item Example code:
    \begin{verbatim}
\documentclass{article}
\usepackage{amsmath}
\begin{document}
\begin{equation*}
  1 + 2 = 3 
\end{equation*}
\begin{equation*}
  1 = 3 - 2
\end{equation*}
\begin{align*}
  1 + 2 &= 3\\
  1 &= 3 - 2
\end{align*}
\end{document}
    \end{verbatim}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Fractions and More}
  \begin{itemize}
    \item LaTeX can typeset integrals, fractions, and more
    \item Example code:
    \begin{verbatim}
\documentclass{article}
\usepackage{amsmath}
\begin{document}
\begin{align*}
  f(x) &= x^2\\
  g(x) &= \frac{1}{x}\\
  F(x) &= \int^a_b \frac{1}{3}x^3
\end{align*}
\end{document}
    \end{verbatim}
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Matrices}
  \begin{itemize}
    \item Display matrices using the \texttt{matrix} environment
    \item Matrices work within math environments
    \item Example code:
    \begin{verbatim}
\begin{matrix}
1 & 0\\
0 & 1
\end{matrix}
    \end{verbatim}
    \item Output: $\begin{matrix}1 & 0\\0 & 1\end{matrix}$
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Brackets in Math Mode - Scaling}
  \begin{itemize}
    \item Plain brackets don't scale with matrix size
    \item Use \texttt{\textbackslash{}left[} and \texttt{\textbackslash{}right]} to scale brackets
    \item Example code:
    \begin{verbatim}
\left[
\begin{matrix}
1 & 0\\
0 & 1
\end{matrix}
\right]
    \end{verbatim}
    \item Output: $\left[\begin{matrix}1 & 0\\0 & 
        1 & 0\
        0 & 1
        \end{matrix}\right]$
        \end{itemize}
        \end{frame}
        
        \begin{frame}[fragile]{Conclusion}
        \begin{itemize}
        \item LaTeX is great for typesetting math and equations
        \item Use the \texttt{equation} and \texttt{align} environments for single and multiple equations
        \item Use \texttt{matrix} environment for matrices
        \item Brackets can be scaled with \texttt{\textbackslash{}left} and \texttt{\textbackslash{}right}
        \end{itemize}
        \end{frame}
