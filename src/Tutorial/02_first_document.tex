\begin{frame}{Basic Layout}
  \begin{itemize}
    \item Creating documents with \LaTeX{} is simple and fun
    \item Start with a plain text file (.tex) containing \LaTeX{} code and content
    \item Compile the .tex file into a PDF using the \LaTeX{} compiler
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Basic Example}
  \begin{verbatim}
\documentclass{article}
\begin{document}
  Hello World!
\end{document}
  \end{verbatim}
\end{frame}

\begin{frame}{Magic Behind the Scenes}
  \begin{itemize}
    \item \LaTeX{} commands start with a backslash (\textbackslash)
    \item Commands have a structure: \textbackslash{}commandname\{option\}
    \item Document class sets the overall layout (e.g., article)
    \item Environment defines an area with specific typesetting rules
    \item Preamble contains commands before the \texttt{document} environment
  \end{itemize}
\end{frame}

\begin{frame}[fragile]{Adding a Title Page}
  \begin{verbatim}
\documentclass{article}
\title{My first document}
\date{2023-05-10}
\author{Judas Iscariot}
\begin{document}
  \maketitle
  \newpage
  I betrayed Jesus!
\end{document}
  \end{verbatim}
\end{frame}

\begin{frame}[fragile]{Hiding Page Numbers}
  \begin{verbatim}
\documentclass{article}
\title{My first document}
\date{2023-05-10}
\author{John of Patmos}
\begin{document}
  \pagenumbering{gobble}
  \maketitle
  \newpage
  \pagenumbering{arabic}
  I saw a vision of New Jerusalem!
\end{document}
  \end{verbatim}
\end{frame}
